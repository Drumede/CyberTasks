}

double Circle::CalArea() {
	double area = 3.14*radius*radius;
	
	return area;
	
}

double Circle:: getRad() {
	return radius;
}


double Circle::CalCircumference() {
	double circumference = 2 * (3.14)*radius;
	if (circumference < 0)
	{
		throw ShapeException();
	}
	return circumference;


}
